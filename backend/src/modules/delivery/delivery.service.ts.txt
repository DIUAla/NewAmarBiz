// backend/src/modules/delivery/delivery.service.ts
import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import axios from 'axios';
import { Order, OrderStatus } from '../../entities/order.entity';

@Injectable()
export class DeliveryService {
  constructor(
    @InjectRepository(Order)
    private orderRepository: Repository<Order>,
  ) {}

  async requestDelivery(orderId: string, userId: string, courierName: string): Promise<Order> {
    const order = await this.orderRepository.findOne({
      where: { id: orderId, userId },
    });

    if (!order) {
      throw new NotFoundException('Order not found');
    }

    // Integration with Pathao/Steadfast API
    let trackingNumber = '';
    let courierResponse = {};

    try {
      switch (courierName.toLowerCase()) {
        case 'pathao':
          courierResponse = await this.requestPathaoDelivery(order);
          trackingNumber = courierResponse['tracking_id'];
          break;
        case 'steadfast':
          courierResponse = await this.requestSteadfastDelivery(order);
          trackingNumber = courierResponse['tracking_code'];
          break;
        default:
          trackingNumber = `TRK-${Date.now()}-${Math.floor(Math.random() * 1000)}`;
      }
    } catch (error) {
      console.error('Courier API error:', error);
      // Fallback: generate manual tracking
      trackingNumber = `MANUAL-${Date.now()}-${Math.floor(Math.random() * 1000)}`;
    }

    order.courierName = courierName;
    order.trackingNumber = trackingNumber;
    order.status = OrderStatus.SHIPPED;

    await this.orderRepository.save(order);

    return order;
  }

  async trackDelivery(trackingNumber: string, userId: string): Promise<any> {
    const order = await this.orderRepository.findOne({
      where: { trackingNumber, userId },
    });

    if (!order) {
      throw new NotFoundException('Order not found');
    }

    // Simulate tracking status
    return {
      trackingNumber: order.trackingNumber,
      status: order.status,
      courier: order.courierName,
      estimatedDelivery: new Date(Date.now() + 3 * 24 * 60 * 60 * 1000),
      history: [
        { status: 'Order confirmed', timestamp: order.createdAt },
        { status: 'Order shipped', timestamp: order.updatedAt },
      ],
    };
  }

  private async requestPathaoDelivery(order: Order): Promise<any> {
    // Pathao API integration
    const response = await axios.post(
      'https://api.pathao.com/aladdin/api/v1/orders',
      {
        merchant_order_id: order.orderNumber,
        recipient_name: order.shippingAddress.name,
        recipient_phone: order.shippingAddress.phone,
        recipient_address: order.shippingAddress.address,
        delivery_type: 48,
        item_type: 1,
        item_quantity: order.items.reduce((sum, item) => sum + item.quantity, 0),
        item_weight: 1000,
        price: order.totalAmount,
      },
      {
        headers: {
          Authorization: `Bearer ${process.env.PATHAO_API_TOKEN}`,
          'Content-Type': 'application/json',
        },
      },
    );
    return response.data;
  }

  private async requestSteadfastDelivery(order: Order): Promise<any> {
    // Steadfast API integration
    const response = await axios.post(
      'https://portal.steadfast.com.bd/api/v1/create_order',
      {
        invoice: order.orderNumber,
        recipient_name: order.shippingAddress.name,
        recipient_phone: order.shippingAddress.phone,
        recipient_address: order.shippingAddress.address,
        cod_amount: order.totalAmount,
      },
      {
        headers: {
          'Api-Key': process.env.STEADFAST_API_KEY,
          'Secret-Key': process.env.STEADFAST_SECRET_KEY,
          'Content-Type': 'application/json',
        },
      },
    );
    return response.data;
  }
}
