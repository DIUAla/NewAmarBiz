// frontend/components/FacebookIntegration.tsx
'use client';

import { useState, useEffect } from 'react';
import { useAuth } from '@/hooks/useAuth';
import { FacebookIcon, RefreshIcon, SettingsIcon, ChartBarIcon } from '@heroicons/react/24/outline';

interface FacebookPage {
  id: string;
  facebookPageId: string;
  name: string;
  category: string;
  isActive: boolean;
  settings: {
    autoCreateOrders: boolean;
    autoReplyComments: boolean;
    replyTemplate: string;
    keywords: string[];
  };
  lastSyncAt: string;
}

interface FacebookComment {
  id: string;
  fromName: string;
  message: string;
  extractedData: {
    productName: string | null;
    quantity: number;
    phone: string | null;
    confidence: number;
  };
  status: string;
  commentedAt: string;
}

export default function FacebookIntegration() {
  const { token } = useAuth();
  const [pages, setPages] = useState<FacebookPage[]>([]);
  const [selectedPage, setSelectedPage] = useState<FacebookPage | null>(null);
  const [comments, setComments] = useState<FacebookComment[]>([]);
  const [loading, setLoading] = useState(false);
  const [syncing, setSyncing] = useState(false);
  const [analytics, setAnalytics] = useState<any>(null);

  useEffect(() => {
    fetchPages();
  }, []);

  const fetchPages = async () => {
    try {
      const response = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/pages`, {
        headers: { Authorization: `Bearer ${token}` },
      });
      const data = await response.json();
      setPages(data.pages || []);
    } catch (error) {
      console.error('Error fetching pages:', error);
    }
  };

  const connectFacebook = () => {
    window.location.href = `${process.env.NEXT_PUBLIC_API_URL}/facebook/auth-url?redirectUri=${encodeURIComponent(window.location.origin + '/facebook/callback')}`;
  };

  const syncPage = async (pageId: string) => {
    setSyncing(true);
    try {
      const response = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/pages/${pageId}/sync`, {
        method: 'POST',
        headers: { Authorization: `Bearer ${token}` },
      });
      const result = await response.json();
      alert(`Synced ${result.newComments} new comments`);
      if (selectedPage?.id === pageId) {
        fetchComments(pageId);
      }
    } catch (error) {
      console.error('Error syncing:', error);
    } finally {
      setSyncing(false);
    }
  };

  const fetchComments = async (pageId: string) => {
    setLoading(true);
    try {
      const response = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/pages/${pageId}/comments`, {
        headers: { Authorization: `Bearer ${token}` },
      });
      const data = await response.json();
      setComments(data.comments || []);
    } catch (error) {
      console.error('Error fetching comments:', error);
    } finally {
      setLoading(false);
    }
  };

  const processComment = async (commentId: string, action: 'create_order' | 'ignore') => {
    try {
      const response = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/comments/${commentId}/process`, {
        method: 'POST',
        headers: {
          Authorization: `Bearer ${token}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ action }),
      });
      const data = await response.json();
      if (data.comment) {
        fetchComments(selectedPage!.id);
      }
    } catch (error) {
      console.error('Error processing comment:', error);
    }
  };

  const toggleAutoSync = async (pageId: string, enable: boolean) => {
    try {
      if (enable) {
        await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/pages/${pageId}/auto-sync`, {
          method: 'POST',
          headers: {
            Authorization: `Bearer ${token}`,
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({ intervalMinutes: 15 }),
        });
      } else {
        await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/pages/${pageId}/auto-sync`, {
          method: 'DELETE',
          headers: { Authorization: `Bearer ${token}` },
        });
      }
      alert(enable ? 'Auto-sync enabled' : 'Auto-sync disabled');
    } catch (error) {
      console.error('Error toggling auto-sync:', error);
    }
  };

  const fetchAnalytics = async (pageId: string) => {
    try {
      const response = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/facebook/pages/${pageId}/analytics?days=30`, {
        headers: { Authorization: `Bearer ${token}` },
      });
      const data = await response.json();
      setAnalytics(data);
    } catch (error) {
      console.error('Error fetching analytics:', error);
    }
  };

  const selectPage = (page: FacebookPage) => {
    setSelectedPage(page);
    fetchComments(page.id);
    fetchAnalytics(page.id);
  };

  return (
    <div className="bg-white rounded-xl shadow-sm">
      {/* Header */}
      <div className="p-6 border-b border-gray-200">
        <div className="flex items-center justify-between">
          <div className="flex items-center space-x-3">
            <div className="w-10 h-10 bg-blue-600 rounded-lg flex items-center justify-center">
              <FacebookIcon className="w-5 h-5 text-white" />
            </div>
            <div>
              <h2 className="text-xl font-semibold text-gray-900">Facebook Integration</h2>
              <p className="text-sm text-gray-500">Sync comments and auto-create orders</p>
            </div>
          </div>
          {pages.length === 0 && (
            <button
              onClick={connectFacebook}
              className="px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 transition"
            >
              Connect Facebook
            </button>
          )}
        </div>
      </div>

      {/* Pages List */}
      {pages.length > 0 && (
        <div className="border-b border-gray-200">
          <div className="p-4">
            <h3 className="text-sm font-medium text-gray-700 mb-3">Connected Pages</h3>
            <div className="space-y-2">
              {pages.map((page) => (
                <div
                  key={page.id}
                  onClick={() => selectPage(page)}
                  className={`flex items-center justify-between p-3 rounded-lg cursor-pointer transition ${
                    selectedPage?.id === page.id
                      ? 'bg-blue-50 border border-blue-200'
                      : 'hover:bg-gray-50 border border-transparent'
                  }`}
                >
                  <div className="flex items-center space-x-3">
                    <div className="w-8 h-8 bg-gray-100 rounded-full flex items-center justify-center">
                      <FacebookIcon className="w-4 h-4 text-blue-600" />
                    </div>
                    <div>
                      <p className="font-medium text-gray-900">{page.name}</p>
                      <p className="text-xs text-gray-500">{page.category}</p>
                    </div>
                  </div>
                  <div className="flex items-center space-x-2">
                    <button
                      onClick={(e) => {
                        e.stopPropagation();
                        syncPage(page.id);
                      }}
                      disabled={syncing}
                      className="p-1 text-gray-400 hover:text-gray-600"
                    >
                      <RefreshIcon className={`w-4 h-4 ${syncing ? 'animate-spin' : ''}`} />
                    </button>
                  </div>
                </div>
              ))}
            </div>
          </div>
        </div>
      )}

      {/* Selected Page Details */}
      {selectedPage && (
        <div className="p-6">
          {/* Settings Toggle */}
          <div className="mb-6 p-4 bg-gray-50 rounded-lg">
            <div className="flex items-center justify-between">
              <div className="flex items-center space-x-2">
                <SettingsIcon className="w-4 h-4 text-gray-500" />
                <span className="text-sm font-medium">Auto-create orders from comments</span>
              </div>
              <button
                onClick={() => toggleAutoSync(selectedPage.id, !selectedPage.settings?.autoCreateOrders)}
                className={`relative inline-flex h-6 w-11 items-center rounded-full transition ${
                  selectedPage.settings?.autoCreateOrders ? 'bg-blue-600' : 'bg-gray-300'
                }`}
              >
                <span
                  className={`inline-block h-4 w-4 transform rounded-full bg-white transition ${
                    selectedPage.settings?.autoCreateOrders ? 'translate-x-6' : 'translate-x-1'
                  }`}
                />
              </button>
            </div>
            <p className="text-xs text-gray-500 mt-2">
              Automatically create orders when customers comment with keywords
            </p>
          </div>

          {/* Analytics Summary */}
          {analytics && (
            <div className="grid grid-cols-3 gap-4 mb-6">
              <div className="text-center p-3 bg-green-50 rounded-lg">
                <p className="text-2xl font-bold text-green-600">{analytics.summary?.convertedOrders || 0}</p>
                <p className="text-xs text-gray-600">Orders Created</p>
              </div>
              <div className="text-center p-3 bg-blue-50 rounded-lg">
                <p className="text-2xl font-bold text-blue-600">{analytics.summary?.totalComments || 0}</p>
                <p className="text-xs text-gray-600">Total Comments</p>
              </div>
              <div className="text-center p-3 bg-purple-50 rounded-lg">
                <p className="text-2xl font-bold text-purple-600">{analytics.summary?.conversionRate || 0}%</p>
                <p className="text-xs text-gray-600">Conversion Rate</p>
              </div>
            </div>
          )}

          {/* Comments List */}
          <div>
            <div className="flex items-center justify-between mb-4">
              <h3 className="font-medium text-gray-900">Pending Comments</h3>
              <button
                onClick={() => syncPage(selectedPage.id)}
                disabled={syncing}
                className="text-sm text-blue-600 hover:text-blue-700"
              >
                {syncing ? 'Syncing...' : 'Sync Now'}
              </button>
            </div>

            {loading ? (
              <div className="space-y-3">
                {[1, 2, 3].map((i) => (
                  <div key={i} className="animate-pulse">
                    <div className="h-20 bg-gray-100 rounded-lg"></div>
                  </div>
                ))}
              </div>
            ) : comments.length === 0 ? (
              <div className="text-center py-8 text-gray-500">
                <p>No pending comments</p>
                <p className="text-sm">Sync your page to fetch comments</p>
              </div>
            ) : (
              <div className="space-y-3">
                {comments.map((comment) => (
                  <div key={comment.id} className="border rounded-lg p-4">
                    <div className="flex items-start justify-between">
                      <div className="flex-1">
                        <div className="flex items-center space-x-2 mb-2">
                          <span className="font-medium text-gray-900">{comment.fromName}</span>
                          <span className="text-xs text-gray-500">
                            {new Date(comment.commentedAt).toLocaleString()}
                          </span>
                        </div>
                        <p className="text-gray-700 mb-2">{comment.message}</p>
                        {comment.extractedData && comment.extractedData.confidence > 0 && (
                          <div className="bg-gray-50 rounded p-2 text-sm">
                            <p className="text-gray-600">
                              🎯 AI Confidence: {(comment.extractedData.confidence * 100).toFixed(0)}%
                            </p>
                            {comment.extractedData.productName && (
                              <p>📦 Product: {comment.extractedData.productName}</p>
                            )}
                            {comment.extractedData.quantity > 1 && (
                              <p>🔢 Quantity: {comment.extractedData.quantity}</p>
                            )}
                            {comment.extractedData.phone && (
                              <p>📞 Phone: {comment.extractedData.phone}</p>
                            )}
                          </div>
                        )}
                      </div>
                      <div className="flex space-x-2 ml-4">
                        <button
                          onClick={() => processComment(comment.id, 'create_order')}
                          className="px-3 py-1 bg-green-600 text-white text-sm rounded hover:bg-green-700"
                        >
                          Create Order
                        </button>
                        <button
                          onClick={() => processComment(comment.id, 'ignore')}
                          className="px-3 py-1 bg-gray-300 text-gray-700 text-sm rounded hover:bg-gray-400"
                        >
                          Ignore
                        </button>
                      </div>
                    </div>
                  </div>
                ))}
              </div>
            )}
          </div>
        </div>
      )}
    </div>
  );
}
