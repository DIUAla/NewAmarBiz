// backend/src/modules/facebook/facebook.module.ts
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';
import { FacebookController } from './facebook.controller';
import { FacebookService } from './facebook.service';
import { FacebookPage } from '../../entities/facebook-page.entity';
import { FacebookComment } from '../../entities/facebook-comment.entity';
import { Order } from '../../entities/order.entity';
import { BullModule } from '@nestjs/bull';

@Module({
  imports: [
    TypeOrmModule.forFeature([FacebookPage, FacebookComment, Order]),
    BullModule.registerQueue({
      name: 'facebook-sync',
    }),
  ],
  controllers: [FacebookController],
  providers: [FacebookService],
  exports: [FacebookService],
})
export class FacebookModule {}
